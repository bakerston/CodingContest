## 1905. Count Sub Islands
## BFS
```swift
def countSubIslands(self, A1: List[List[int]], A2: List[List[int]]) -> int:
        m, n = len(A1), len(A1[0])
        dq = collections.deque()
        dirs = ((1, 0), (0, 1), (-1, 0), (0, -1))
        idx = 2
        for i in range(m):
            for j in range(n):
                if A1[i][j] == 1:
                    A1[i][j] = idx
                    dq.append((i, j))
                    while dq:
                        cx, cy = dq.popleft()
                        for dx, dy in dirs:
                            x, y = cx + dx, cy + dy
                            if 0 <= x < m and 0 <= y < n and A1[x][y] == 1:
                                dq.append((x, y))
                                A1[x][y] = idx
                    idx += 1
        
        for i in range(m):
            for j in range(n):
                if A2[i][j] == 1:
                    A2[i][j] = idx
                    dq.append((i, j))
                    while dq:
                        cx, cy = dq.popleft()
                        for dx, dy in dirs:
                            x, y = cx + dx, cy + dy
                            if 0 <= x < m and 0 <= y < n and A2[x][y] == 1:
                                dq.append((x, y))
                                A2[x][y] = idx
                    idx += 1
        pos = collections.defaultdict(set)
        for i in range(m):
            for j in range(n):
                if A2[i][j] > 1:
                    pos[A2[i][j]].add(A1[i][j])
        ans = 0
        for key in pos:
            if len(pos[key]) == 1 and 0 not in pos[key]:
                ans += 1
        return ans
```
