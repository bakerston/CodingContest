## 1898. Maximum Number of Removable Characters
```swift
def maximumRemovals(self, s: str, p: str, R: List[int]) -> int:
        n = len(R)
        m = len(s)
        lp = len(p)

        def issub(idx):
            rset = set(R[:idx])
            tmp = ""
            for i in range(m):
                if i not in rset:
                    tmp += s[i]
            i, j = 0, 0
            while i < lp and j < len(tmp): 
                if p[i] == tmp[j]:
                    i += 1
                    j += 1
                else:
                    j += 1
            return i == lp

        l, r = 0, n
        while l < r:
            mid = r - (r - l) // 2
            if issub(mid):
                l = mid
            else:
                r = mid - 1
        return l
```
