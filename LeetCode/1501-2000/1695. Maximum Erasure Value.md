## 1695. Maximum Erasure Value
## Hashtable
```swift
def maximumUniqueSubarray(self, nums: List[int]) -> int:
        d, pre, ans = {}, 0, 0
        presum = [0] + list(itertools.accumulate(nums))

        for i, x in enumerate(nums):
            if x not in d:
                d[x] = i             
            else:
                pre = max(pre, d[x] + 1)
                d[x] = i
            ans = max(ans, presum[i + 1] - presum[pre])
            
        return ans
```
