## 1918. Kth Smallest Subarray Sum
## Binary Search
## Python
```swift
def kthSmallestSubarraySum(self, nums: List[int], k: int) -> int:
        acc = [0] + list(itertools.accumulate(nums))
        n = len(nums)
        
        def bsh(target):  # arr numbers when sum(arr) <= target
            ans = 0
            for i in range(n):
                cur = acc[i] + target
                j = bisect.bisect_right(acc, cur)
                ans += j - i - 1
            return ans
               
        l, r = 0, sum(nums)
        
        while l < r:
            m = (l + r) // 2
            if bsh(m) < k:
                l = m + 1
            else:
                r = m
        
        return l
```
