## 1881. Maximum Value after Insertion
```swift
def maxValue(self, n: str, x: int) -> str:
        if n[0] == "-":
            le = len(n)
            for i in range(1, le):
                if x < int(n[i]):
                    return n[:i] + str(x) + n[i:]
            return n + str(x)
        else:
            le = len(n)
            for i in range(le):
                if x > int(n[i]):
                    return n[:i] + str(x) + n[i:]
            return n + str(x)
```
