## 1930. Unique Length-3 Palindromic Subsequences
## Python
```swift
def countPalindromicSubsequence(self, s: str) -> int:
        c = collections.Counter(s)
        ans = 0
        for ch in c.keys():
            l, r = s.find(ch), s.rfind(ch)
            if l == r or l + 1 == r:
                continue
            cur = collections.Counter(s[l + 1: r])
            ans += len(cur.keys())
        return ans
```
