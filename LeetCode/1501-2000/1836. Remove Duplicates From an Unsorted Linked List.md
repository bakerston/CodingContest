## 1836. Remove Duplicates From an Unsorted Linked List
```swift
def deleteDuplicatesUnsorted(self, head: ListNode) -> ListNode:
        p = ListNode(0)
        p.next = head
        p = p.next
        
        d = collections.defaultdict(int)
      
        while p:
            d[p.val] += 1
            p = p.next

        r = q = ListNode(0)
        q.next = head
        while q.next:
            if d[q.next.val] > 1:
                q.next = q.next.next
            else:
                q = q.next
             
        return r.next
```
