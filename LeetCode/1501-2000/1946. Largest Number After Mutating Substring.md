## 1946. Largest Number After Mutating Substring
## Python
```swift
def maximumNumber(self, num: str, change: List[int]) -> str:
        ans, n, idx = "", len(num), 0
        while idx < n and change[int(num[idx])] <= int(num[idx]):
            ans += num[idx]
            idx += 1
        while idx < n and change[int(num[idx])] >= int(num[idx]):
            ans += str(change[int(num[idx])])
            idx += 1
        while idx < n:
            ans += num[idx]
            idx += 1
        return ans
```
