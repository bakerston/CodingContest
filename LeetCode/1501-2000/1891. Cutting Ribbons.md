## 1891. Cutting Ribbons
~~~SQL
def maxLength(self, ribbons: List[int], k: int) -> int:
        def get_num(length):
            return sum(map(lambda x: x // length, ribbons)) >= k
        if sum(ribbons) < k:
            return 0
        l, r = 1, max(ribbons)
        while l < r:
            m = r - (r - l) // 2
            if get_num(m):
                l = m
            else:
                r = m - 1
        return l
~~~
