## 1615. Maximal Network Rank
```swift
def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
      degrees = [0] * n
      graph = collections.defaultdict(set)
      for i, j in roads:
          degrees[i] += 1
          degrees[j] += 1
          graph[i].add(j)
          graph[j].add(i)
      ans = 0

      for i in range(n - 1):
          for j in range(i + 1, n):
              ans = max(ans, degrees[i] + degrees[j] - int(j in graph[i]))
      return ans
```
