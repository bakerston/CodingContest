## 1863. Sum of All Subset XOR Totals
```swift
def subsetXORSum(self, nums: List[int]) -> int:
        def getor(lst):
            n = len(lst)
            res = lst[0]
            for i in lst[1:]:
                res ^= i
            return res
        ans = 0
        n = len(nums)
        for i in range(2, n + 1):
            c = list(itertools.combinations(nums, i))
            for k in c:
                ans += getor(k)
        return ans + sum(nums)
 ```
