## 1899. Merge Triplets to Form Target Triplet
```swift
def mergeTriplets(self, T: List[List[int]], t: List[int]) -> bool:
        res = [0, 0, 0]
        n = len(T)
        for i in range(n):
            if T[i][0] <= t[0] and T[i][1] <= t[1] and T[i][2] <= t[2]:
                res[0] = max(res[0], T[i][0])
                res[1] = max(res[1], T[i][1])
                res[2] = max(res[2], T[i][2])
        return res == t
```
