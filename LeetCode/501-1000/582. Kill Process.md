## 582. Kill Process
## BFS
```swift
def killProcess(self, pid: List[int], ppid: List[int], kill: int) -> List[int]:
        nxt, dq = collections.defaultdict(set), collections.deque([kill])
        ans, n = [], len(pid)
        
        for i in range(n):
            nxt[ppid[i]].add(pid[i])

        while dq:
            cur = dq.popleft()
            ans.append(cur)            
            for cand in nxt[cur]:
                dq.append(cand)
                
        return ans
```
