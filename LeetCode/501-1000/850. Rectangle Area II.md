## 850. Rectangle Area II
```swift
def rectangleArea(self, R: List[List[int]]) -> int:
        vert, mod = [], 10 ** 9 + 7
        for i, x in enumerate(R):
            vert.append(x[1])
            vert.append(x[3])            
        vert = sorted(list(set(vert)))
        
        n, idx, prel = len(vert), 0, vert[0]       
        xx, yy = [], []
        
        while idx < n:
            curl, tmp = vert[idx], []
            for x in R:
                if x[1] <= curl < x[3]:
                    tmp.append([x[0], x[2]])
            idx += 1
            tmp.sort()
            res, lft, rgt = 0, 0, 0
            for l, r in tmp:
                if l >= rgt:
                    res += r - l
                else:
                    res += max(0, r - rgt)
                rgt = max(r, rgt)
            xx.append(res)
            yy.append(curl - prel)
            prel = curl

        return sum(x * y for x, y in zip(xx, yy[1:])) % mod
```
