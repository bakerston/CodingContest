def threeEqualParts(self, arr: List[int]) -> List[int]:
        m = sum(arr)
        if m % 3:
            return [-1, -1]
        k = m // 3
        i = 0
        n = len(arr)
        if m == 0:
            return [0, n - 1]
        
        #start with finding the third (last) binary value in the list
        for idx in range(n):
            if arr[idx] == 1:
                i += 1
            if i == 2 * k + 1:
                res = arr[idx :]
                break
        reslen = len(res)
        
        ans = []
        
        #find the first binary value in list
        for idx in range(n):
            if arr[idx] == 1:
                break
        if arr[idx : idx + reslen] == res:
            ans.append(idx + reslen - 1)
        else:
            return [-1, -1]

        #find the second binary in the list
        for idx in range(idx + reslen, n):
            if arr[idx] == 1:
                break
        if arr[idx : idx + reslen] == res:
            ans.append(idx + reslen)
            return ans
        else:
            return [-1, -1]
