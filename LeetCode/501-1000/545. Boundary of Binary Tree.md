## 545. Boundary of Binary Tree
## 三次recurse，第一次preorder找左边界，第二次找叶结点，第三次postorder找右边界。
```swift
def boundaryOfBinaryTree(self, root: TreeNode) -> List[int]:
        def lft(root):
            if (not root) or (not root.left and not root.right):
                return
            ans.append(root.val)
            if root.left:
                lft(root.left)
            else:
                lft(root.right)
        
        def rgt(root):
            if not root or (not root.left and not root.right):
                return
            if root.right:
                rgt(root.right)
            else:
                rgt(root.left)
            ans.append(root.val)
        
        def leaves(root):
            if not root:
                return
            leaves(root.left)
            if not root.left and not root.right:
                ans.append(root.val)
            leaves(root.right)
        
        if not root:
            return []
        if not root.left and not root.right:
            return [root.val]
        
        ans = [root.val]
        lft(root.left)
        leaves(root)
        rgt(root.right)
        return ans
```
