 ## 656. Coin Path
 ## DP
 ```swift
 def cheapestJump(self, coins: List[int], jp: int) -> List[int]:
      n = len(coins)
      cost = [0] + [math.inf] * (n - 1)
      path = [[0]] + [[]] * (n - 1)

      for i in range(1, n):
          if coins[i] < 0:
              continue
          for j in range(max(0, i - jp), i):
              if cost[i] > cost[j] + coins[j]:
                  cost[i] = cost[j] + coins[j]
                  path[i] = path[j] + [i]
              elif cost[i] == cost[j] + coins[j]:
                  if path[i] > path[j] + [i]:
                      path[i] = path[j] + [i]

      if not path[n - 1]:
          return []
      else:
          return list(map(lambda x: x + 1, path[n - 1]))
 ```
