## 915. Partition Array into Disjoint Intervals
## Python
```swift
def partitionDisjoint(self, A: List[int]) -> int:
        n = len(A)
        lft, rgt = [A[0]] + [0] * (n - 1), [math.inf] * (n - 1) + [A[-1]]
        for i in range(1, n):
            lft[i] = max(lft[i - 1], A[i])
            rgt[-1 - i] = min(rgt[-i], A[-i])
        for i in range(n):
            if lft[i] <= rgt[i]:
                return i + 1
```
