## 838. Push Dominoes
## Python
```swift
def pushDominoes(self, D: str) -> str:
        idx, n, ans = 0, len(D), []
        while idx < n:
            if D[idx] in "LR":
                ans.append(D[idx])
                idx += 1
            else:
                s = idx
                while idx < n and D[idx] == ".":
                    idx += 1
                e = idx # idx - 1
                k = e - s
                if s == 0 and e == n:
                    cur = ["."] * k
                elif s == 0:
                    cur = ["."] * k if D[e] == "R" else ["L"] * k
                elif e == n:
                    cur = ["."] * k if D[s - 1] == "L" else ["R"] * k
                else:
                    if D[s - 1] == "R" and D[e] == "L":
                        cur = ["R"] * (k // 2) + ["."] * (k % 2) + ["L"] * (k // 2)
                    elif D[s - 1] == "R":
                        cur = ["R"] * k
                    elif D[e] == "L":
                        cur = ["L"] * k
                    else:
                        cur = ["."] * k
                ans.extend(cur)
        return "".join(ans)
```
