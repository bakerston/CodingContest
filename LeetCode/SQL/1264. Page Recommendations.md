## 1264. Page Recommendations
~~~SQL
SELECT DISTINCT page_id AS recommended_page 
FROM
(
SELECT user1_id f
FROM Friendship
WHERE user2_id = 1
UNION
SELECT user2_id 
FROM Friendship
WHERE user1_id = 1
) a
LEFT JOIN Likes b
ON a.f = b.user_id

WHERE page_id NOT IN
(
SELECT page_id
    FROM Likes
    WHERE user_id = 1
) 
AND page_id IS NOT NULL
~~~
