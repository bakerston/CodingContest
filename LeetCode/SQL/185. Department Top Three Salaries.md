## 185. Department Top Three Salaries
## DENSE_RANK
~~~sql
SELECT d.Name Department, m.Name Employee, Salary
FROM
(
SELECT DepartmentId, Name, Salary, 
DENSE_RANK() OVER
(PARTITION BY DepartmentId ORDER BY Salary DESC) AS rk
FROM Employee
) m,
Department d
WHERE m.DepartmentId = d.Id
AND rk < 4
~~~
