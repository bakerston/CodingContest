## 1096. Brace Expansion II
## DFS 傻逼问题，给爷做吐了

def braceExpansionII(self, E: str) -> List[str]:
        E = "{" + E + "}"
        getr, st = collections.defaultdict(int), []
        for i, x in enumerate(E):
            if x == "{": st.append(i)
            if x == "}": getr[st.pop()] = i

        def unbrack(lft, rgt):
            ans = []
            ii = lft + 1
            while ii < rgt:
                if E[ii] == "{":
                    rr = getr[ii]
                    cur = unbrack(ii, rr)
                    if not ans or ans[-1] == " ":
                        ans.append(cur)
                    else:
                        pre = ans.pop()
                        new = list(itertools.product(pre, cur))
                        new = ["".join(x) for x in new]
                        ans.append(new)
                    ii = rr + 1
                elif E[ii].isalpha():
                    if ans and ans[-1] != " ":
                        ans[-1] = [x + E[ii] for x in ans[-1]]
                    else:
                        ans.append([E[ii]])                 
                    ii += 1
                elif E[ii] == ",":
                    ans.append(" ")
                    ii += 1
            i, res = 0, []
            cur = []
            while i < len(ans):
                if ans[i] == " ":
                    i += 1
                    continue
                while i < len(ans) and ans[i] != " ":
                    cur.append(tuple(ans[i]))
                    i += 1
                res.extend(list(itertools.product(*cur)))
                i += 1
                cur = []
            ans = []
            for x in res:
                for y in x:
                    ans.append(y)
            return sorted(list(set(ans)))
        
        ans = []
        ii = 0
        while ii < len(E):
            if E[ii] == "{":
                ans.append(unbrack(ii, getr[ii]))
                ii = getr[ii] + 1
            else:
                if not ans:
                    ans.append([E[ii]])
                else:
                    ans[-1] = [x + E[ii] for x in ans[-1]]              
                ii += 1
        ans = list(itertools.product(*ans))
        ans = ["".join(x) for x in ans]
        return sorted(list(set(ans)))
