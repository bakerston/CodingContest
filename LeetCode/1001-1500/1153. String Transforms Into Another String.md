## 1153. String Transforms Into Another String

```swift
def canConvert(self, str1: str, str2: str) -> bool:
        same, d = 0, collections.defaultdict()
        for i, j in zip(str1, str2):
            if i in d:
                if j != d[i]:
                    return False
            else:
                d[i] = j

        for i in d:
            same += i == d[i]
        return True if len(set(d.values())) < 26 else same == 26

```
