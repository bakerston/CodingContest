## 1171. Remove Zero Sum Consecutive Nodes from Linked List
## Python
```swift
def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        dummy = ListNode(0)
        pre = 0
        prenode = {}
        prenode[0] = dummy
        dummy.next = head
        
        while head:
            pre += head.val
            prenode[pre] = head
            head = head.next
        head = dummy
        pre = 0
        while head:
            pre += head.val
            head.next = prenode[pre].next
            head = head.next
        return dummy.next
```
