## 1008. Construct Binary Search Tree from Preorder Traversal
## Stack
## Python
```swift
def bstFromPreorder(self, A: List[int]) -> TreeNode:
        root = TreeNode(A[0])
        st = [root]
        
        for val in A[1:]:
            if val < st[-1].val:
                last = st[-1]
                last.left = TreeNode(val)
                st.append(last.left)
            else:
                while st and st[-1].val < val:
                    last = st.pop()
                last.right = TreeNode(val)
                st.append(last.right)
        return root
```
