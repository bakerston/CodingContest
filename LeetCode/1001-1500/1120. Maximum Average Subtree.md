## 1120. Maximum Average Subtree
## DFS
```swift
def maximumAverageSubtree(self, root: TreeNode) -> float:
        self.ans = 0
        def dfs(root):
            if not root:
                return [0, 0]
            ls, ln = dfs(root.left)
            rs, rn = dfs(root.right)
            self.ans = max(self.ans, float((ls + rs + root.val)/(ln + rn + 1)))
            return [ls + rs + root.val, ln + rn + 1]
        dfs(root)
        
        return self.ans
```
