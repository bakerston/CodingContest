## 464. Can I Win
## DFS 
```swift
def canIWin(self, M: int, D: int) -> bool:
        memo = {}
        t = list(range(1, M + 1))
        if sum(t) == D:
            return len(t) % 2 == 1
        if t[-1] >= D:
            return True
        if sum(t) < D:
            return False
        
        
        def dfs(t, res):
            tu = tuple(t)
            if t[-1] >= res:
                return True
            if tu in memo:
                return memo[tu]
            
            for i in range(len(t)):
                if not dfs(t[:i] + t[i + 1:], res - t[i]):
                    memo[tu] = True
                    return True
            memo[tu] = False
            return False
        
        return dfs(t, D)
```
