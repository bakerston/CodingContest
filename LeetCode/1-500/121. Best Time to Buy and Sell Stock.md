## 121. Best Time to Buy and Sell Stock
## Python
 def maxProfit(self, A: List[int]) -> int:
        ans, buy = 0, math.inf
        for a in A:
            ans = max(ans, a - buy)
            buy = min(buy, a)
        return ans
