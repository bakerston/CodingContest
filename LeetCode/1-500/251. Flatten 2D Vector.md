```swift
class Vector2D:

    def __init__(self, vec: List[List[int]]):
        self.vec = vec
        self.inner = 0
        self.outer = 0
       
    def next(self) -> int:
        while self.inner == len(self.vec[self.outer]):
            self.inner = 0
            self.outer += 1
        ans = self.vec[self.outer][self.inner]
        self.inner += 1
        return ans

    def hasNext(self) -> bool:
        while self.outer < len(self.vec):
            if self.inner < len(self.vec[self.outer]):
                return True
            self.inner = 0
            self.outer += 1
        return False
```

## CPP
```swift
class Vector2D {
    int row;
    int col;
    vector<vector<int>> data;
    
public:
    Vector2D(vector<vector<int>>& vec) {
        data = vec;
        row = 0;
        col = 0;
    }
    
    int next() {
        while(col == data[row].size()){
            col = 0;
            row += 1;
        }
        int ans = data[row][col];
        col += 1;
        return ans;
    }
    
    bool hasNext() {
        while(row < data.size()){
            if (col < data[row].size()){
                return true;
            }
            row += 1;
            col = 0;
        }
        return false;
    }
};
```
