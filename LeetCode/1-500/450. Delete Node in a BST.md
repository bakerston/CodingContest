## 450. Delete Node in a BST
## Python
```swift
def deleteNode(self, root: TreeNode, key: int) -> TreeNode:
        if not root:
            return root
        if root.val > key:
            root.left = self.deleteNode(root.left, key)
        elif root.val < key:
            root.right = self.deleteNode(root.right, key)
        else:
            if not root.right:
                return root.left
            if not root.left:
                return root.right
            tmp = root.right
            val = tmp.val
            while tmp.left:
                tmp = tmp.left
                val = tmp.val
            root.val = val
            root.right = self.deleteNode(root.right, val)
        return root
```
