## 236. Lowest Common Ancestor of a Binary Tree
#### 1.记录每个结点和其左右结点的状态，如果左节点有```p, q```中任意一个，则有```l = 1```，如果根结点为```p, q```中任意一个
####    则有```m = 1```

```swift
def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
  def dfs(root):
      if not root:
          return False
      l = dfs(root.left)
      r = dfs(root.right)

      m = root  == p or root == q
      if m + l + r >= 2:
          self.ans = root

      return m or l or r
  dfs(root)
  return self.ans
```

#### 2.DFS
```
def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
      st = [root]
      parent = {root: None}

      while p not in parent or q not in parent:
          node = st.pop()
          if node.left:
              parent[node.left] = node
              st.append(node.left)
          if node.right:
              parent[node.right] = node
              st.append(node.right)
      anc = set()
      while p:
          anc.add(p)
          p = parent[p]
      while q not in anc:
          q = parent[q]
      return q
```
