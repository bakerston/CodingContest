## 95. Unique Binary Search Trees II
## Python
```swift
def generateTrees(self, n: int) -> List[TreeNode]:
        def helper(lft, rgt):
            if lft > rgt:
                return [None]
            ans = []
            for i in range(lft, rgt + 1):
                lt = helper(lft, i - 1)
                rt = helper(i + 1, rgt)
                
                for l in lt:
                    for r in rt:
                        cur = TreeNode(i)
                        cur.left = l
                        cur.right = r
                        ans.append(cur)
            return ans
        return helper(1, n) if n else []
```
